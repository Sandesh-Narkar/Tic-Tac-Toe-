The JS Logic

Turn Tracking
let turn0 = true;

if (turn0) {
  box.innerText = "O";
  turn0 = false;
} else {
  box.innerText = "X";
  turn0 = true;
}
Win Patterns (8 total)
const winPatterns = [
  [0,1,2],
  [3,4,5],
  [6,7,8],
  [0,3,6],
  [1,4,7],
  [2,5,8],
  [0,4,8],
  [2,4,6],
];
Check Winner
for (let p of winPatterns) {
  let a = boxes[p[0]].innerText;
  let b = boxes[p[1]].innerText;
  let c = boxes[p[2]].innerText;

  if(a !== "" && a===b && b===c){
    showWinner(a);
    return;
  }
}
Draw Detection
let full = [...boxes]
  .every(b => b.innerText !== "");

if (full) showDraw();
